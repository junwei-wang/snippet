* Add Certificate into Java Keystore

  - Download certificate
    #+BEGIN_SRC sh
    echo -n | openssl s_client -connect itebeta.baidu.com:443 | sed -ne '/-BEGIN CERTIFICATE-/,/-END CERTIFICATE-/p' > a.cert
    #+END_SRC

  - Add certificate into keystore
    #+BEGIN_SRC sh
    echo -n | openssl s_client -connect itebeta.baidu.com:443 | sed -ne '/-BEGIN CERTIFICATE-/,/-END CERTIFICATE-/p' > a.cert
    #+END_SRC
